CREATE TABLE `menu` (

  `menuName` varchar(16) DEFAULT NULL COMMENT '菜单名字',

  `menuIcon` varchar(100) DEFAULT NULL,

CREATE TABLE `record` (
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT '主键',

  `userId` int(11) DEFAULT NULL COMMENT '取货人/补货人',
